package aliounebadara;

import java.util.Collections;

public class AliouneBadara {

    /**
     * @param args the command line arguments
     */
    public int InversionInt(ArrayList<Integer> arrayInt) {
        Collections.sort(arrayInt, Collections.reverseOrder());
        int y = 0;
        for (int i : arrayInt) {
            y = i;
            System.out.println(i);
        }
        return y;
    }

    public float InversFloat(ArrayList<Float> arrayFlaot) {
        Collections.sort(arrayFlaot, Collections.reverseOrder());
        float f = 0;
        for (Float float1 : arrayFlaot) {
            f = float1;
            System.out.println(float1);
        }
        return f;
    }

    public String InversionString(ArrayList<String> arraylist) {
        Collections.sort(arraylist, Collections.reverseOrder());
        String str2 = null;
        for (String str : arraylist) {
            str2 = str;
            System.out.println(str);
        }

        return str2;
    }

        System.out.println("Inversion Int");
        AliouneBadara i = new AliouneBadara();

        ArrayList<Float> arrayfloat = new ArrayList<Float>();
        arrayfloat.add(0.0f);
        arrayfloat.add(8.2f);
        arrayfloat.add(3.5f);
        arrayfloat.add(5.0f);
        System.out.println("Inversion Int");
        AliouneBadara f = new AliouneBadara();
        f.InversFloat(arrayfloat);
        //Inversion Float
